✨ كل شيء جاهز! اتبع الخطوات التالية:

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

🎯 الطريقة 1: OpenAI (مباشرة)

   OPENAI_API_KEY="sk-your-openai-key" docker compose up --build -d

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

🎯 الطريقة 2: OpenRouter (موصى به)

   OPENAI_API_KEY="sk-or-v1-your-key" \
   API_BASE_URL="https://openrouter.ai/api/v1" \
   MODEL="openai/gpt-3.5-turbo" \
   docker compose up --build -d

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

🎯 الطريقة 3: كل المتغيرات معاً

   OPENAI_API_KEY="your-key" \
   API_BASE_URL="https://openrouter.ai/api/v1" \
   MODEL="openai/gpt-3.5-turbo" \
   PORT=8080 \
   docker compose up --build -d

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

🌐 بعد التشغيل:

   • افتح: http://localhost:8080
   • API Docs: http://localhost:8080/docs
   • Health: http://localhost:8080/health

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📋 الأوامر الأساسية:

   عرض السجلات:
   $ docker compose logs -f

   إيقاف التطبيق:
   $ docker compose down

   إعادة التشغيل:
   $ docker compose restart

   حالة الـ containers:
   $ docker compose ps

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

⚠️ ملاحظات مهمة:

   1. لا تضع API Key في أي ملف
   2. استخدم متغيرات البيئة فقط
   3. .env موجود في .gitignore
   4. يمكنك استخدام .env.example كمرجع
