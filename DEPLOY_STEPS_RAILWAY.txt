⚠️ مهم جداً: API Key يجب أن يبقى مخفي!

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📋 الخطوة 1: حذف API Key القديم وإنشاء واحد جديد

   ⚠️ المفتاح الذي أرسلته أصبح مكشوف!
   
   1. اذهب إلى: https://openrouter.ai/keys
   2. احذف المفتاح القديم (sk-or-v1-5f460201...)
   3. اضغط "Create Key"
   4. انسخ المفتاح الجديد
   5. احتفظ به في مكان آمن
   6. لا تشاركه مع أحد!

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📋 الخطوة 2: رفع المشروع على GitHub

   افتح terminal في مجلد المشروع:

   git init
   git add .
   git commit -m "Square B Chatbot - Initial commit"
   git branch -M main
   
   # أنشئ repository جديد على GitHub:
   # https://github.com/new
   # ثم:
   
   git remote add origin https://github.com/YOUR_USERNAME/square-b-chatbot.git
   git push -u origin main

   ✅ المشروع الآن على GitHub بدون أي API Keys!

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📋 الخطوة 3: إنشاء حساب على Railway

   1. اذهب إلى: https://railway.app
   2. اضغط "Login"
   3. سجّل دخول بـ GitHub
   4. امنح Railway الصلاحيات

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📋 الخطوة 4: نشر المشروع على Railway

   1. في Railway Dashboard، اضغط "New Project"
   
   2. اختر "Deploy from GitHub repo"
   
   3. اختر repository: square-b-chatbot
   
   4. Railway سيبدأ البناء تلقائياً
   
   5. انتظر حتى يكتمل البناء (2-3 دقائق)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📋 الخطوة 5: إضافة API Key بشكل آمن ومخفي

   في Railway Dashboard:
   
   1. اضغط على اسم المشروع
   
   2. اذهب إلى تبويب "Variables"
   
   3. اضغط "New Variable"
   
   4. أضف هذه المتغيرات واحداً تلو الآخر:
   
      Variable Name: OPENAI_API_KEY
      Value: sk-or-v1-YOUR-NEW-KEY-HERE
      (استخدم المفتاح الجديد من الخطوة 1)
   
      Variable Name: MODEL
      Value: openai/gpt-3.5-turbo
   
      Variable Name: API_BASE_URL
      Value: https://openrouter.ai/api/v1
   
   5. اضغط "Add" لكل متغير

   ✅ الآن API Key مخفي وآمن!

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📋 الخطوة 6: إعادة النشر

   بعد إضافة المتغيرات:
   
   1. اذهب إلى "Deployments"
   
   2. اضغط "Redeploy" أو انتظر النشر التلقائي
   
   3. انتظر 1-2 دقيقة

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📋 الخطوة 7: الحصول على رابط الموقع

   1. في Railway Dashboard
   
   2. اذهب إلى "Settings" → "Domains"
   
   3. اضغط "Generate Domain"
   
   4. ستحصل على رابط مثل:
      https://square-b-chatbot-production.railway.app

   ✅ هذا هو رابط موقعك!

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📋 الخطوة 8: تجربة الموقع

   1. افتح الرابط في المتصفح
   
   2. ستظهر واجهة Square B Chatbot الوردية
   
   3. جرّب إرسال رسالة مثل:
      "مرحبا"
      "كم سعر برجر بيف؟"
      "شو بتنصحني؟"
   
   4. إذا رد البوت → كل شيء يعمل! ✅

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📋 الخطوة 9: التحقق من Logs

   إذا لم يعمل البوت:
   
   1. في Railway Dashboard
   
   2. اذهب إلى "Deployments"
   
   3. اضغط "View Logs"
   
   4. ابحث عن أي أخطاء
   
   5. تأكد أن API_KEY صحيح
